import React, { useCallback, useMemo, useState } from 'react'
import * as XLSX from 'xlsx'
 
type ParsedWorkbook = {
  sheets: Record<string, Array<Record<string, any>>>
  sheetNames: string[]
}
 
const EXPECTED_SHEETS = [
  'Dados gerais',
  'Equipe do projeto',
  'Cliente',
  'Produto contratado',
  'Documentos chaves',
  'Riscos',
  'Cronograma macro',
  'Cronograma por vertical'
]
 
export default function Importer() {
  const [fileName, setFileName] = useState<string>('')
  const [data, setData] = useState<ParsedWorkbook | null>(null)
  const [error, setError] = useState<string | null>(null)
  const [isParsing, setIsParsing] = useState<boolean>(false)
 
  const handleFiles = useCallback(async (file: File) => {
    setError(null)
    setIsParsing(true)
    try {
      const buffer = await file.arrayBuffer()
      const wb = XLSX.read(buffer, { type: 'array' })
 
      const parsed: ParsedWorkbook = { sheets: {}, sheetNames: wb.SheetNames }
      wb.SheetNames.forEach((name) => {
        const ws = wb.Sheets[name]
        if (!ws) return
        const rows = XLSX.utils.sheet_to_json(ws, { defval: '' }) as Array<Record<string, any>>
        parsed.sheets[name] = rows
      })
 
      setFileName(file.name)
      setData(parsed)
    } catch (e: any) {
      setError(e?.message || 'Falha ao ler a planilha.')
    } finally {
      setIsParsing(false)
    }
  }, [])
 
  const onInputChange = useCallback((e: React.ChangeEvent<HTMLInputElement>) => {
    const f = e.target.files?.[0]
    if (f) handleFiles(f)
  }, [handleFiles])
 
  const onDrop = useCallback((e: React.DragEvent<HTMLDivElement>) => {
    e.preventDefault()
    const f = e.dataTransfer.files?.[0]
    if (f) handleFiles(f)
  }, [handleFiles])
 
  const onDragOver = useCallback((e: React.DragEvent<HTMLDivElement>) => {
    e.preventDefault()
  }, [])
 
  const missingSheets = useMemo(() => {
    if (!data) return EXPECTED_SHEETS
    const present = new Set(data.sheetNames.map((n) => n.toLowerCase()))
    return EXPECTED_SHEETS.filter((s) => !present.has(s.toLowerCase()))
  }, [data])
 
  const handleSaveLocal = useCallback(() => {
    if (!data) return
    try {
      const payload = {
        imported_at: new Date().toISOString(),
        file_name: fileName,
        data
      }
      window.localStorage.setItem('importer_last_payload', JSON.stringify(payload))
      alert('Dados da planilha salvos no navegador (localStorage).')
    } catch (e: any) {
      setError(e?.message || 'Falha ao salvar localmente.')
    }
  }, [data, fileName])
 
  return (
    <div className="max-w-5xl mx-auto py-10">
      <h1 className="text-3xl font-bold text-secondary-900 dark:text-secondary-100 mb-6">Importador de Planilhas (XLSX)</h1>
      <p className="text-secondary-700 dark:text-secondary-300 mb-6">
        Arraste e solte uma planilha XLSX ou selecione um arquivo para importar. Serão lidas as abas esperadas do protótipo: Dados gerais, Equipe do projeto, Cliente, Produto contratado, Documentos chaves, Riscos, Cronograma macro e por vertical.
      </p>
 
      <div
        onDrop={onDrop}
        onDragOver={onDragOver}
        className="border-2 border-dashed rounded-lg p-8 text-center bg-white dark:bg-secondary-800 border-secondary-300 dark:border-secondary-600"
      >
        <input
          type="file"
          accept=".xlsx,.xls"
          onChange={onInputChange}
          aria-label="Selecionar arquivo XLSX"
          className="hidden"
          id="file-input"
        />
        <label htmlFor="file-input" className="cursor-pointer inline-block bg-primary-600 hover:bg-primary-700 text-white px-4 py-2 rounded">
          Selecionar arquivo
        </label>
        <div className="mt-3 text-sm text-secondary-600 dark:text-secondary-300">ou arraste e solte aqui</div>
      </div>
 
      {isParsing && (
        <div className="mt-6 text-sm opacity-80">Lendo planilha...</div>
      )}
 
      {error && (
        <div className="mt-6 p-3 rounded border border-red-300 text-red-700 bg-red-50">{error}</div>
      )}
 
      {data && (
        <div className="mt-8 space-y-6">
          <div className="flex items-center justify-between">
            <div>
              <div className="text-sm text-secondary-500 dark:text-secondary-400">Arquivo</div>
              <div className="font-medium text-secondary-900 dark:text-secondary-100">{fileName}</div>
            </div>
            <button
              type="button"
              onClick={handleSaveLocal}
              className="bg-emerald-600 hover:bg-emerald-700 text-white px-4 py-2 rounded"
            >
              Salvar no navegador
            </button>
          </div>
 
          <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
            <div className="card p-4">
              <h3 className="font-semibold mb-2">Abas encontradas</h3>
              <ul className="text-sm list-disc pl-5 space-y-1">
                {data.sheetNames.map((n) => (
                  <li key={n}>{n}</li>
                ))}
              </ul>
            </div>
 
            <div className="card p-4">
              <h3 className="font-semibold mb-2">Abas esperadas ausentes</h3>
              {missingSheets.length === 0 ? (
                <div className="text-sm text-emerald-600">Todas as abas esperadas foram encontradas.</div>
              ) : (
                <ul className="text-sm list-disc pl-5 space-y-1">
                  {missingSheets.map((n) => (
                    <li key={n}>{n}</li>
                  ))}
                </ul>
              )}
            </div>
          </div>
 
          {/* Pré-visualização de até 5 linhas por aba */}
          <div className="space-y-6">
            {data.sheetNames.map((name) => {
              const rows = data.sheets[name] || []
              const headers = rows.length > 0 ? Object.keys(rows[0]) : []
              const preview = rows.slice(0, 5)
              return (
                <div key={name} className="card p-4 overflow-x-auto">
                  <div className="font-semibold mb-2">
                    {name} <span className="text-xs text-secondary-500">({rows.length} linhas)</span>
                  </div>
                  {headers.length === 0 ? (
                    <div className="text-sm opacity-70">Sem dados nesta aba.</div>
                  ) : (
                    <table className="min-w-full text-sm">
                      <thead>
                        <tr>
                          {headers.map((h) => (
                            <th key={h} className="text-left font-medium px-3 py-2 border-b">{h}</th>
                          ))}
                        </tr>
                      </thead>
                      <tbody>
                        {preview.map((r, i) => (
                          <tr key={i} className="border-b last:border-0">
                            {headers.map((h) => (
                              <td key={h} className="px-3 py-2">{String(r[h] ?? '')}</td>
                            ))}
                          </tr>
                        ))}
                      </tbody>
                    </table>
                  )}
                </div>
              )
            })}
          </div>
        </div>
      )}
    </div>
  )
}

import React, { useEffect, useMemo, useState } from 'react'
import { Link } from 'react-router-dom'
import { useProjects } from '../api/projects'
 
export default function ProjectsList() {
  const { projects, loading, error, fetchProjects } = useProjects()
 
  const [search, setSearch] = useState('')
  const [status, setStatus] = useState('')
  const [portfolio, setPortfolio] = useState('')
  const [page, setPage] = useState(1)
  const pageSize = 10
 
  useEffect(() => {
    fetchProjects()
  }, [fetchProjects])
 
  const filtered = useMemo(() => {
    const text = search.trim().toLowerCase()
    return projects.filter(p => {
      const matchesText = !text || (p.name?.toLowerCase().includes(text) || p.description?.toLowerCase().includes(text))
      const matchesStatus = !status || p.status === status
      const matchesPortfolio = !portfolio || (p.portfolio === portfolio)
      return matchesText && matchesStatus && matchesPortfolio
    })
  }, [projects, search, status, portfolio])
 
  const totalPages = Math.max(1, Math.ceil(filtered.length / pageSize))
  const currentPage = Math.min(page, totalPages)
  const paginated = useMemo(() => {
    const start = (currentPage - 1) * pageSize
    return filtered.slice(start, start + pageSize)
  }, [filtered, currentPage])
 
  useEffect(() => {
    // Reset para primeira página ao mudar filtros/busca
    setPage(1)
  }, [search, status, portfolio])
 
  if (loading) {
    return (
      <div>
        <h3>Projetos</h3>
        <div className="flex justify-center items-center h-64">
          <div className="animate-spin rounded-full h-32 w-32 border-b-2 border-primary-600"></div>
        </div>
      </div>
    )
  }
 
  if (error) {
    return (
      <div className="bg-red-50 border border-red-200 rounded-md p-4">
        <p className="text-red-800">Erro ao carregar projetos: {error}</p>
        <button 
          onClick={() => fetchProjects()}
          className="mt-2 px-4 py-2 bg-red-600 text-white rounded hover:bg-red-700"
        >
          Tentar novamente
        </button>
      </div>
    )
  }
 
  return (
    <div>
      <h3>Projetos</h3>
 
      <div className="mt-4 mb-4 grid grid-cols-1 md:grid-cols-4 gap-3" aria-label="filtros">
        <input
          aria-label="buscar projetos"
          type="text"
          placeholder="Buscar por nome ou descrição"
          className="border rounded px-3 py-2"
          value={search}
          onChange={e => setSearch(e.target.value)}
        />
 
        <select
          aria-label="filtrar por status"
          className="border rounded px-3 py-2"
          value={status}
          onChange={e => setStatus(e.target.value)}
        >
          <option value="">Todos os status</option>
          <option value="not_started">Não iniciado</option>
          <option value="on_track">No prazo</option>
          <option value="warning">Atenção</option>
          <option value="delayed">Atrasado</option>
          <option value="completed">Concluído</option>
        </select>
 
        <input
          aria-label="filtrar por portfólio"
          type="text"
          placeholder="Portfólio"
          className="border rounded px-3 py-2"
          value={portfolio}
          onChange={e => setPortfolio(e.target.value)}
        />
      </div>
 
      <ul role="list">
        {paginated.map(p => (
          <li key={p.id} className="py-1">
            <Link to={`/projects/${p.id}`}>{p.name}</Link>
          </li>
        ))}
      </ul>
 
      <div className="flex items-center justify-between mt-4" aria-label="paginação">
        <button
          onClick={() => setPage(p => Math.max(1, p - 1))}
          disabled={currentPage === 1}
          className="px-3 py-1 border rounded disabled:opacity-50"
          aria-label="página anterior"
        >
          Anterior
        </button>
        <span className="text-sm" aria-live="polite">
          Página {currentPage} de {totalPages}
        </span>
        <button
          onClick={() => setPage(p => Math.min(totalPages, p + 1))}
          disabled={currentPage === totalPages}
          className="px-3 py-1 border rounded disabled:opacity-50"
          aria-label="próxima página"
        >
          Próxima
        </button>
      </div>
    </div>
  )
}

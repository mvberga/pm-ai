import React from 'react'
import { useProjects } from '../../api/projects'
import Checklist from '../../components/Checklist'
 
export default function ChecklistsPage() {
  const { projects, fetchProjects, loading, error } = useProjects()
  const [projectId, setProjectId] = React.useState('')
 
  React.useEffect(() => {
    fetchProjects()
  }, [fetchProjects])
 
  const selectedId = projectId || (projects[0]?.id?.toString?.() ?? '')
 
  return (
    <div className="space-y-4">
      <h3 className="text-xl font-semibold">Checklists</h3>
      <p className="opacity-80">Selecione um projeto para gerenciar grupos e itens.</p>
 
      {loading && <div>Carregando projetos…</div>}
      {error && <div role="alert" className="text-red-600">{error}</div>}
 
      <div className="flex items-center gap-2">
        <label className="text-sm opacity-80">Projeto:</label>
        <select
          value={selectedId}
          onChange={(e) => setProjectId(e.target.value)}
          className="border rounded px-2 py-1"
        >
          {projects.map((p) => (
            <option key={p.id} value={p.id}>{p.name}</option>
          ))}
        </select>
      </div>
 
      {selectedId ? (
        <Checklist projectId={selectedId} />
      ) : (
        <div className="opacity-70">Nenhum projeto disponível.</div>
      )}
    </div>
  )
}
